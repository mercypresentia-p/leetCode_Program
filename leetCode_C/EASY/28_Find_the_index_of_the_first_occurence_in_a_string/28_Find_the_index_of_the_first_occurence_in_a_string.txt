int strStr(char* haystack, char* needle) {
    int i = 0;
    while (i < (strlen(haystack) / strlen(needle))){
        int c = 0;
        for (int j = 0; j < strlen(needle); j++){
            if (haystack[i + j] == needle[j])
                c++;
        }
        if (c == strlen(needle)){
            return i;
        }
        i++;
    }
    return -1;
}
